NOTICE테이블 생성
CREATE TABLE NOTICE (
    NNUM VARCHAR2(20) PRIMARY KEY,
    NSUBJECT VARCHAR2(200) NOT NULL,
    NWRITER VARCHAR2(100) NOT NULL,
    NCONTENTS VARCHAR2(500),
    NFILE VARCHAR2(200),
    DELETEYN VARCHAR2(1) NOT NULL,
    INSERTDATE DATE,
    UPDATEDATE DATE
);
