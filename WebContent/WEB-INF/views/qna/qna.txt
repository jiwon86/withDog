--Q&A 테이블 생성
CREATE TABLE QNA (
    QNANUM VARCHAR2(30) PRIMARY KEY,
    QNATITLE VARCHAR2(200),
    QNACON VARCHAR2(500),
    QNAWRITER VARCHAR2(50),
    QNAPW VARCHAR2(4),
    QNAANSWER VARCHAR2(1),
    QNAFILE VARCHAR2(200),
    QNAINSERTDATE DATE,
    QNAUPDATEDATE DATE,
    DELYN VARCHAR2(1)
    );

CREATE TABLE SE_QNA(
    QNANUM VARCHAR2(30) PRIMARY KEY,
    QNATITLE VARCHAR2(200),
    QNACON VARCHAR2(500),
    QNAWRITER VARCHAR2(50),
    QNAPW VARCHAR2(4),
    QNAANSWER VARCHAR2(1),
    QNAFILE VARCHAR2(200),
    QNAINSERTDATE DATE,
    QNAUPDATEDATE DATE,
    DELYN VARCHAR2(1)
    );

CREATE SEQUENCE QNA_SEQ START WITH 1 INCREMENT BY 1 ;
